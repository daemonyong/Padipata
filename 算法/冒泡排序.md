冒泡排序

```java


public class BubblingSort {
    public static void main(String[] args) {
        int[] arr = {124,465,98,365,78,55};
        for (int i = 0; i < arr.length-1 ; i++) {//-1为了防止下标越界
            for (int j = 0; j < arr.length-1-i ; j++) {// -1-i是为了减少比较的次数，每次都会确定一个最大值。
                if(arr[j]>arr[j+1]){
                    int t = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = t;
                }
            }
        }
        for (int i : arr) {
            System.out.println(i);
        }
    }
}
```
